# Linux Debian 环境部署

jdk版本17.0.13
```
jdk安装详见：https://blog.csdn.net/phker/article/details/128078431
```

maven版本3.8.7

```
配置文件：src/main/resources/application.yaml
部署前需要修改以下配置项：

--- 数据库 ---
spring.datasource.url / username / password

--- 邮箱 ---
my-mail.to 收件邮箱地址
spring.mail.username 发件邮箱地址
spring.mail.password 发件邮箱授权码

--- 微信支付 ---
car.wechat.appid / secret / mchid / mchSerialNo / apiV3Key
car.wechat.privateKeyFilePath 商户私钥文件路径
car.wechat.weChatPayCertFilePath 微信支付平台证书文件路径
car.wechat.notifyUrl / refundNotifyUrl 支付回调地址

--- JWT ---
car.jwt.admin-secret-key / car.jwt.user-secret-key

--- 云存储 ---
cloud.aliyun.oss.access-key-id / access-key-secret / bucket-name
cloud.tencent.cos.secret-id / secret-key / bucket-name / base-url

--- 其他 ---
convexHull.host 凸包计算服务地址
```

```yaml
server:
  port: 9090 # 端口
my-mail:
  to: xxxxxxxx@xx.com # 收件邮箱地址
spring:
  mail:
    host: smtp.163.com
    port: 465
    username: your-email@example.com # 发件邮箱地址
    password: your-email-auth-code # 发件邮箱授权码
    protocol: smtp
    properties:
      mail:
        smtp:
          auth: true
          ssl:
            enable: true
          starttls:
            enable: false
          socketFactory:
            class: javax.net.ssl.SSLSocketFactory
    default-encoding: UTF-8
  jackson:
    date-format: yyyy/MM/dd HH:mm
    time-zone: GMT+8
  datasource: # 数据库连接
    driver-class-name: com.mysql.cj.jdbc.Driver
    url: jdbc:mysql://your-db-host:3306/your_database?serverTimezone=Asia/Shanghai
    username: your_db_username
    password: your_db_password
    type: com.alibaba.druid.pool.DruidDataSource
mybatis:
  configuration:
    map-underscore-to-camel-case: true

car:
  wechat:
    appid: wx_your_appid_here # 小程序appid
    secret: your_app_secret_here # 小程序secret
    mchid: 1234567890 # 商户号
#    mchSerialNo: YOUR_OLD_SERIAL_NO # 证书序列号(旧)
    mchSerialNo: YOUR_MCH_SERIAL_NO # 证书序列号
    privateKeyFilePath: /path/to/your/apiclient_key.pem # 私钥文件 替换为服务器实际文件地址
    apiV3Key: your_api_v3_key_here # 解密密钥
    weChatPayCertFilePath: /path/to/your/wechatpay_cert.pem # 平台证书文件 替换为服务器实际文件地址
    notifyUrl: https://your-domain.com/notify/paySuccess # 支付成功回调
    refundNotifyUrl: https://your-domain.com/notify/refundSuccess # 支付退款回调
  jwt:
    # 设置jwt签名加密时使用的秘钥
    admin-secret-key: your_jwt_secret_key
    # 设置jwt过期时间
    admin-ttl: 720000000000
    # 设置前端传递过来的令牌名称
    admin-token-name: authentication
    user-secret-key: your_jwt_secret_key
    # token过期时间
    user-ttl: 720000000000
    user-token-name: authentication
# springdoc-openapi项目配置
springdoc:
  swagger-ui:
    path: /swagger-ui.html
    tags-sorter: alpha
    operations-sorter: alpha
  api-docs:
    path: /v3/api-docs
  group-configs:
    - group: 'default'
      paths-to-match: '/**'
      packages-to-scan: com.sunl19ht.client.controller
# knife4j的增强配置，不需要增强可以不配
knife4j:
  enable: true
  setting:
    language: zh-CN
convexHull:
  host: http://your-convex-hull-service:8080 # nodejs后端服务地址，用于计算区域价格 需要单独运行node

my-mail:
  to: your-receiver@example.com # 收件邮箱地址

cloud:
  aliyun:
    oss:
      endpoint: https://oss-cn-shanghai.aliyuncs.com
      access-key-id: your_access_key_id
      access-key-secret: your_access_key_secret
      bucket-name: your-bucket-name
  tencent:
    cos:
      secret-id: your_secret_id
      secret-key: your_secret_key
      region: ap-shanghai
      bucket-name: your-bucket-name-appid
      base-url: https://your-bucket.cos.ap-shanghai.myqcloud.com/
```

```shell
git clone -b develop git@your-git-server:your-org/your-repo.git
cd car-rental--java-server
mvn clean package # JAR包 /car-rental--java-server/target/car-rental-1.0-SNAPSHOT.jar
nohup java -jar car-rental-1.0-SNAPSHOT.jar > logfile.log 2>&1 & # 后台运行
cat logfile.log # 日志文件
```
